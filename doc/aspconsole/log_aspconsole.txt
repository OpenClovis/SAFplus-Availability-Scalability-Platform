/**
 * \addtogroup log_aspconsole Log Service ASP Console Commands
 * \{
 */

/**
*****************************************
\brief Writes a log record.
 
\par Usage: 
\code writeLog <handle> <severity> <message> \endcode 

\param handle[DEC] Stream handle, 1 for  CL_LOG_HANDLE_SYS  or 2 for CL_LOG_HANDLE_APP  or handle returned from  clLogOpen()
\param severity[DEC] Severity of the message {1-8},1 being highest.
\param message[STRING] The log message.

\par Output:
\code 
writeLog 1 4 test-message
Message logged successfully 
\endcode 
***************************************** 
*/
writeLog;
 
/**
*****************************************
\brief Opens an application stream for logging.
 
\par Usage: 
\code openStream <recSize> <fileSize> <fileName> <fileLoc> \endcode

\param recSize[DEC] Max Record Size
\param fileSize[DEC] Max File Size
\param fileName[STRING] Name of file
\param fileLoc[STRING] File Location

\par Output:
\code 
openStream 100 1000 test1 /home/
Successfully opened a stream, handle returned [4] 
\endcode 
***************************************** 
*/
openStream;

/**
*****************************************
\brief Closes an application stream.
 
\par Usage: 
\code closeStream <handle> \endcode
        
\param handle[DEC] handle to the stream

\par Output:
\code 
closeStream 4
Successfully closed a stream with handle [4] 
\endcode 
***************************************** 
*/
closeStream;


/**
*****************************************
\brief Setting systemlog level
 
\par Usage: 
\code sysLvlSet nodeAddress severity \endcode

\param nodeAddress[HEX] The nodeAddress where severity is to be set
\param severity[DEC] The severity to be set

\par Output:
\code 
sysLvlSet 0xbb 1
Successfully set system wide log level  
\endcode 
***************************************** 
*/
sysLvlSet;
